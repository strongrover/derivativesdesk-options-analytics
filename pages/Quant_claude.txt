# ── Core Math Engine ──────────────────────────────────────────────────────────
class BlackScholes:
    """Full Black-Scholes pricing + Greeks engine."""

    @staticmethod
    def d1d2(S, K, T, sigma, r, q=0.0):
        if T <= 0 or sigma <= 0:
            return None, None
        d1 = (np.log(S / K) + (r - q + 0.5 * sigma ** 2) * T) / (sigma * np.sqrt(T))
        d2 = d1 - sigma * np.sqrt(T)
        return d1, d2

    @classmethod
    def price(cls, S, K, T, sigma, r, option_type="call", q=0.0):
        d1, d2 = cls.d1d2(S, K, T, sigma, r, q)
        if d1 is None:
            return max(S - K, 0) if option_type == "call" else max(K - S, 0)
        if option_type == "call":
            return S * np.exp(-q * T) * norm.cdf(d1) - K * np.exp(-r * T) * norm.cdf(d2)
        else:
            return K * np.exp(-r * T) * norm.cdf(-d2) - S * np.exp(-q * T) * norm.cdf(-d1)

    @classmethod
    def greeks(cls, S, K, T, sigma, r, option_type="call", q=0.0):
        d1, d2 = cls.d1d2(S, K, T, sigma, r, q)
        if d1 is None:
            return {g: 0.0 for g in ["delta", "gamma", "theta", "vega", "rho", "vanna", "charm", "volga"]}
        phi = norm.pdf(d1)
        Nd1 = norm.cdf(d1)
        Nd2 = norm.cdf(d2)
        Nd1m = norm.cdf(-d1)
        Nd2m = norm.cdf(-d2)
        sqrtT = np.sqrt(T)

        delta = np.exp(-q * T) * (Nd1 if option_type == "call" else Nd1 - 1)
        gamma = np.exp(-q * T) * phi / (S * sigma * sqrtT)
        theta_call = (
            -np.exp(-q * T) * S * phi * sigma / (2 * sqrtT)
            - r * K * np.exp(-r * T) * Nd2
            + q * S * np.exp(-q * T) * Nd1
        ) / 365
        theta_put = (
            -np.exp(-q * T) * S * phi * sigma / (2 * sqrtT)
            + r * K * np.exp(-r * T) * Nd2m
            - q * S * np.exp(-q * T) * Nd1m
        ) / 365
        theta = theta_call if option_type == "call" else theta_put
        vega = S * np.exp(-q * T) * phi * sqrtT / 100
        rho_call = K * T * np.exp(-r * T) * Nd2 / 100
        rho_put = -K * T * np.exp(-r * T) * Nd2m / 100
        rho = rho_call if option_type == "call" else rho_put

        # Higher-order greeks
        vanna = -np.exp(-q * T) * phi * d2 / sigma
        charm = -np.exp(-q * T) * (
            phi * (2 * (r - q) * T - d2 * sigma * sqrtT) / (2 * T * sigma * sqrtT)
            + (q if option_type == "call" else -q) * Nd1
        )
        volga = vega * d1 * d2 / sigma * 100

        return {
            "delta": delta, "gamma": gamma, "theta": theta,
            "vega": vega, "rho": rho, "vanna": vanna,
            "charm": charm, "volga": volga,
        }

    @classmethod
    def implied_vol(cls, market_price, S, K, T, r, option_type="call", q=0.0):
        """Brent's method implied vol solver."""
        if T <= 0:
            return np.nan
        try:
            intrinsic = max(S - K, 0) if option_type == "call" else max(K - S, 0)
            if market_price <= intrinsic:
                return np.nan
            f = lambda sigma: cls.price(S, K, T, sigma, r, option_type, q) - market_price
            return brentq(f, 1e-6, 10.0, xtol=1e-6, maxiter=500)
        except Exception:
            return np.nan


class PhoenixOption:
    """
    Autocall Phoenix — pays coupon if spot > barrier on observation dates,
    auto-called if spot > call barrier. Capital at risk below put barrier.
    Monte Carlo pricing.
    """

    @staticmethod
    def price(S, K, T, sigma, r, coupon_rate=0.08, call_barrier=1.0,
              put_barrier=0.7, obs_freq=4, n_paths=20000, seed=42):
        np.random.seed(seed)
        dt = T / (obs_freq * T)
        steps = int(obs_freq * T)
        if steps < 1:
            steps = 1
        t_obs = np.linspace(dt, T, steps)

        # GBM paths
        Z = np.random.standard_normal((n_paths, steps))
        log_returns = (r - 0.5 * sigma ** 2) * dt + sigma * np.sqrt(dt) * Z
        S_paths = S * np.exp(np.cumsum(log_returns, axis=1))

        payoffs = np.zeros(n_paths)
        alive = np.ones(n_paths, dtype=bool)

        for i, t in enumerate(t_obs):
            spot = S_paths[:, i]
            # Autocall
            called = alive & (spot >= call_barrier * K)
            payoffs[called] += (1 + coupon_rate * t) * K * np.exp(-r * t)
            alive[called] = False
            # Coupon
            coupon_paid = alive & (spot >= put_barrier * K)
            payoffs[coupon_paid] += coupon_rate * K / obs_freq * np.exp(-r * t)

        # Maturity
        final = S_paths[:, -1]
        # Capital return if above put barrier
        payoffs[alive & (final >= put_barrier * K)] += K * np.exp(-r * T)
        # Capital at risk below put barrier
        payoffs[alive & (final < put_barrier * K)] += final[alive & (final < put_barrier * K)] * np.exp(-r * T)

        return np.mean(payoffs), np.std(payoffs) / np.sqrt(n_paths)


class DigitalOption:
    """Cash-or-nothing digital option."""

    @staticmethod
    def price(S, K, T, sigma, r, option_type="call"):
        if T <= 0 or sigma <= 0:
            return 0.0
        d2 = (np.log(S / K) + (r - 0.5 * sigma ** 2) * T) / (sigma * np.sqrt(T))
        if option_type == "call":
            return np.exp(-r * T) * norm.cdf(d2)
        else:
            return np.exp(-r * T) * norm.cdf(-d2)

# ── Live market data fetcher (cached 5 min) ───────────────────────────────────
@st.cache_data(ttl=300)
def fetch_live_data(ticker_symbol):
    """
    Fetch live spot price, daily change, and realised vol (30d) from Yahoo Finance.
    Returns dict with S, sigma, change_pct, currency, last_updated.
    Falls back to metadata if fetch fails (e.g. market closed, no internet).
    """
    try:
        tkr = yf.Ticker(ticker_symbol)
        hist = tkr.history(period="60d", interval="1d", auto_adjust=True)
        if hist.empty or len(hist) < 5:
            return None

        spot = float(hist["Close"].iloc[-1])
        prev = float(hist["Close"].iloc[-2])
        change_pct = (spot - prev) / prev * 100

        # Realised vol: annualised std of daily log returns over last 30 days
        log_rets = np.log(hist["Close"] / hist["Close"].shift(1)).dropna()
        sigma = float(log_rets.tail(30).std() * np.sqrt(252))

        info = tkr.fast_info
        currency = getattr(info, "currency", "EUR")
        last_updated = hist.index[-1].strftime("%Y-%m-%d %H:%M UTC")

        return {
            "S": spot,
            "sigma": max(sigma, 0.05),   # floor at 5% to avoid degenerate cases
            "change_pct": change_pct,
            "currency": currency,
            "last_updated": last_updated,
            "source": "live",
        }
    except Exception:
        return None


def get_ticker_data(name):
    """Get live data for a ticker, with graceful fallback to hardcoded values."""
    meta = CAC40_META[name]
    live = fetch_live_data(meta["ticker"])
    if live:
        live["sector"] = meta["sector"]
        return live
    # Fallback
    return {
        "S": meta["fallback_S"],
        "sigma": meta["fallback_sigma"],
        "sector": meta["sector"],
        "change_pct": 0.0,
        "currency": "EUR",
        "last_updated": "offline (cached)",
        "source": "fallback",
    }

# TAB 5 — BACKTEST
# ═════════════════════════════════════════════════════════════════════════════
with tabs[4]:
    st.markdown("<div class='section-header'>P&L Backtest — Simulated Price Path</div>", unsafe_allow_html=True)

    col1, col2 = st.columns([1, 3])
    with col1:
        n_sims    = st.slider("Simulated Paths", 100, 5000, 1000, step=100)
        hedge     = st.checkbox("Delta Hedge daily", value=True)
        bt_seed   = st.number_input("Random Seed", 42, 9999, 42)
        show_hist = st.checkbox("Show P&L Distribution", value=True)

    with col2:
        np.random.seed(int(bt_seed))
        dt = 1 / 252
        steps = T_days
        n_show = min(n_sims, 30)

        # GBM simulation
        mu = r - q
        Z = np.random.standard_normal((n_sims, steps))
        log_ret = (mu - 0.5 * sigma ** 2) * dt + sigma * np.sqrt(dt) * Z
        paths = S * np.exp(np.cumsum(log_ret, axis=1))
        paths = np.hstack([np.full((n_sims, 1), S), paths])

        t_axis = np.arange(steps + 1)

        # Final P&L
        if hedge:
            # Delta-hedged P&L
            final_pnls = []
            for p in paths:
                option_pnl = BlackScholes.price(p[-1], K, 0.0001, sigma, r, option_type, q) - ep
                hedge_pnl = -greeks["delta"] * (p[-1] - p[0])
                final_pnls.append((option_pnl + hedge_pnl) * quantity * 100)
        else:
            final_pnls = [(BlackScholes.price(p[-1], K, 0.0001, sigma, r, option_type, q) - ep) * quantity * 100 for p in paths]

        fig_bt = go.Figure()
        for i in range(n_show):
            pnl_path = np.array([
                (BlackScholes.price(paths[i, t], K, max((T_days - t) / 365, 0.0001), sigma, r, option_type, q) - ep) * quantity * 100
                for t in range(steps + 1)
            ])
            color = f"rgba(0,212,255,0.12)" if pnl_path[-1] >= 0 else "rgba(255,51,102,0.12)"
            fig_bt.add_trace(go.Scatter(x=t_axis, y=pnl_path, mode="lines",
                                        line=dict(width=0.8, color=color), showlegend=False))

        # Mean path
        mean_pnl = np.array([
            (BlackScholes.price(np.mean(paths[:, t]), K, max((T_days - t) / 365, 0.0001), sigma, r, option_type, q) - ep) * quantity * 100
            for t in range(steps + 1)
        ])
        fig_bt.add_trace(go.Scatter(x=t_axis, y=mean_pnl, mode="lines",
                                     line=dict(width=2.5, color="#d4c27a"), name="Mean path"))
        fig_bt.add_hline(y=0, line_color="rgba(255,255,255,0.2)")
        apply_layout(fig_bt, f"P&L Evolution — {n_sims} Paths ({'Delta-Hedged' if hedge else 'Unhedged'})",
                     xaxis_title="Days", yaxis_title="P&L (€)")
        st.plotly_chart(fig_bt, use_container_width=True)

    if show_hist:
        col_stats, col_hist = st.columns([1, 2])
        with col_stats:
            pnl_arr = np.array(final_pnls)
            var_95  = np.percentile(pnl_arr, 5)
            var_99  = np.percentile(pnl_arr, 1)
            cvar_95 = pnl_arr[pnl_arr <= var_95].mean()
            win_rate = (pnl_arr > 0).mean() * 100
            st.markdown(f"""
            <div style='background:#080f1c;border:1px solid #0e2040;border-radius:4px;padding:16px;font-family:IBM Plex Mono,monospace;font-size:12px'>
              <div style='color:#4a6a8a;font-size:9px;letter-spacing:2px;text-transform:uppercase;margin-bottom:12px'>Backtest Statistics</div>
              <div style='display:grid;gap:8px'>
                <div><span style='color:#4a6a8a'>Mean P&L</span><br><span style='color:#e8f4ff'>€{pnl_arr.mean():+,.0f}</span></div>
                <div><span style='color:#4a6a8a'>Median P&L</span><br><span style='color:#e8f4ff'>€{np.median(pnl_arr):+,.0f}</span></div>
                <div><span style='color:#4a6a8a'>Std Dev</span><br><span style='color:#e8f4ff'>€{pnl_arr.std():,.0f}</span></div>
                <div><span style='color:#4a6a8a'>VaR 95%</span><br><span style='color:#c87e7e'>€{var_95:,.0f}</span></div>
                <div><span style='color:#4a6a8a'>VaR 99%</span><br><span style='color:#c87e7e'>€{var_99:,.0f}</span></div>
                <div><span style='color:#4a6a8a'>CVaR 95%</span><br><span style='color:#c87e7e'>€{cvar_95:,.0f}</span></div>
